function uploadPhotoToCustomerFolder(
  pelanggan,
  fileName,
  base64Data,
  mimeType
){

  try {

    if(!pelanggan){
      return {
        success:false,
        message:"Nama pelanggan kosong"
      };
    }

    if(!base64Data){
      return {
        success:false,
        message:"Data foto kosong"
      };
    }

    const photoFolder =
      getPhotoFolder(pelanggan);

    // Bersihkan prefix data:image/jpeg;base64,...
    if(base64Data.indexOf(",") > -1){
      base64Data =
        base64Data.split(",")[1];
    }

    const bytes =
      Utilities.base64Decode(base64Data);

    const ext =
      mimeType.includes("png")
      ? "png"
      : "jpg";

    const namaUnik =
      Utilities.formatDate(
        new Date(),
        "Asia/Jakarta",
        "yyyyMMdd_HHmmss"
      ) +
      "_" +
      fileName.replace(/[^\w\-]/g,"_") +
      "." +
      ext;

    const blob =
      Utilities.newBlob(
        bytes,
        mimeType,
        namaUnik
      );

    const file =
      photoFolder.createFile(blob);

    file.setSharing(
      DriveApp.Access.ANYONE_WITH_LINK,
      DriveApp.Permission.VIEW
    );

    return {
      success:true,
      fileId:file.getId(),
      fileName:file.getName(),
      size:file.getSize(),
      url:
        "https://drive.google.com/file/d/" +
        file.getId() +
        "/view"
    };

  } catch(err){

    Logger.log(err);

    return {
      success:false,
      message:err.toString()
    };
  }
}

// 1. FUNGSI UTAMA (Mesin Pemroses Otomatis)
function processTableQR(sheetName, kolomNamaPelanggan, kolomIDTransaksi, prefixNamaFile, kolomTargetDownload, kolomTargetView) {
  let sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(sheetName);
  if (!sheet) return;

  let parentFolder = DriveApp.getFolderById(pelanggan);
  let data = sheet.getDataRange().getValues();

  for (let i = 1; i < data.length; i++) {
    let row = data[i];
    let namaPelanggan = row[kolomNamaPelanggan - 1];
    let idTransaksi = row[kolomIDTransaksi - 1];
    
    // Skrip akan berjalan jika salah satu atau kedua kolom target masih kosong
    let linkDownloadSaatIni = row[kolomTargetDownload - 1];
    let linkViewSaatIni = row[kolomTargetView - 1];

    if (namaPelanggan && idTransaksi && (!linkDownloadSaatIni || !linkViewSaatIni)) {
      try {
        Utilities.sleep(3000); // Jeda 3 detik agar AppSheet selesai upload
        
        let subFolders = parentFolder.getFoldersByName(namaPelanggan);
        if (subFolders.hasNext()) {
          let pelangganFolder = subFolders.next();
          
          let idBersih = String(idTransaksi).trim();
          let namaFileYangDicari = prefixNamaFile + idBersih + ".pdf";
          let files = pelangganFolder.getFilesByName(namaFileYangDicari);

          if (files.hasNext()) {
            let file = files.next();
            let fileId = file.getId();
            
            // Atur izin file agar bisa dilihat oleh siapa saja yang punya link
            try {
              file.setSharing(DriveApp.Access.ANYONE_WITH_LINK, DriveApp.Permission.VIEW);
            } catch(err) {}
            
            // KELUARAN 1: Link khusus Download Langsung (Direct Download)
            let linkDownload = "https://drive.google.com/uc?export=download&id=" + fileId;
            
            // KELUARAN 2: Link khusus View Web (Sangat ramah untuk Scan QR HP)
            let linkView = "https://drive.google.com/file/d/" + fileId + "/view?usp=drivesdk";
            
            // Tulis masing-masing ke kolom targetnya
            sheet.getRange(i + 1, kolomTargetDownload).setValue(linkDownload);
            sheet.getRange(i + 1, kolomTargetView).setValue(linkView);
          }
        }
      } catch (e) {
        Logger.log("Error pada baris " + (i+1) + ": " + e.message);
      }
    }
  }
}

/**
 * Fungsi simpan data yang disesuaikan dengan urutan kolom tabel log:
 * A: key | B: modul_id | C: date | D: PROD_TOTAL | E: BILL_TOTAL | F: prod_harian | G: bill_harian | H: noMesin | I: last_update
 */
function simpanDataDariESP(p) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetPerf = ss.getSheetByName("performa"); 
  const sheetLog = ss.getSheetByName("log"); 
  const stability = p.stability || 0;
  const ppm = p.ppm || 0;
  const sesiDur = p.sesi_dur || "00:00:00";
  const sesiQty = p.sesi_qty || 0;
  //if (!e || !e.parameter) {
    //return ContentService.createTextOutput("ERROR: No Parameter").setMimeType(ContentService.MimeType.TEXT); //[cite: 20]
  //}
  
  //const p = e.parameter;
  const sekarang = new Date();
  
  // Format Tanggal dan Waktu
  const tglID = Utilities.formatDate(sekarang, "GMT+7", "yyyy-MM-dd"); //[cite: 21]
  const fullTime = Utilities.formatDate(sekarang, "GMT+7", "yyyy-MM-dd HH:mm:ss"); //[cite: 21]
  const timestamp = Utilities.formatDate(sekarang, "GMT+7", "yyyy-MM-dd HH:mm:ss"); //[cite: 21]
  
  const snMesin = p.noMesin || p.mesin || p.id; //[cite: 22]
  const keyUnik = tglID + "_" + snMesin; //[cite: 25]

  // --- LOGIKA UPDATE ATAU INSERT KE TABEL 'log' ---
  // Urutan array disesuaikan tepat dengan Kolom A sampai I pada gambar tabel log
  const dataLengkapLog = [
    keyUnik,               // Kolom A: key (Contoh: 2026-05-14_3289M330021) //[cite: 25]
    p.modul_id || "ESP32", // Kolom B: modul_id //[cite: 26]
    tglID,                 // Kolom C: date //[cite: 21]
    p.prod_total || 0,     // Kolom D: # PROD_TOTAL //[cite: 27]
    p.bill_total || 0,     // Kolom E: # BILL_TOTAL //[cite: 28]
    p.prod_harian || 0,    // Kolom F: # prod_harian
    p.bill_harian || 0,    // Kolom G: # bill_harian
    snMesin,               // Kolom H: noMesin //[cite: 22]
    timestamp              // Kolom I: last_update //[cite: 21]
  ];

  const dataLog = sheetLog.getDataRange().getValues(); //[cite: 29]
  let barisDitemukan = -1;

  // Cari apakah Key unik sudah ada di kolom A
  for (let i = 0; i < dataLog.length; i++) {
    if (dataLog[i][0] == keyUnik) {
      barisDitemukan = i + 1; //[cite: 30]
      break; //[cite: 31]
    }
  }

  if (barisDitemukan !== -1) {
    // Jika ada, update baris tersebut //[cite: 32]
    sheetLog.getRange(barisDitemukan, 1, 1, dataLengkapLog.length).setValues([dataLengkapLog]);
  } else {
    // Jika tidak ada, tambah baris baru //[cite: 33]
    sheetLog.appendRow(dataLengkapLog);
  }

  // --- SIMPAN KE TABEL 'performa' ---
  // Sesuai gambar tabel performa: A: KEY (Full Time) | B: NO_MESIN | C: TANGGAL | D: SMALL | dst.
  const dataPerf = [
    fullTime,              // Kolom A: KEY (Timestamp lengkap) //[cite: 34]
    snMesin,               // Kolom B: NO_MESIN //[cite: 34]
    tglID,                 // Kolom C: TANGGAL //[cite: 34]
    parseFloat(p.small) || 0,  // Kolom D: # SMALL //[cite: 34]
    parseFloat(p.large) || 0,  // Kolom E: # LARGE //[cite: 34]
    parseFloat(p.noise) || 0,  // Kolom F: # NOISE //[cite: 34]
    parseFloat(p.jam) || 0,    // Kolom G: # JAM //[cite: 34]
    parseFloat(p.gap) || 0,    // Kolom H: # GAP //[cite: 34]
    parseFloat(p.door) || 0,    // Kolom I: # door //[cite: 34]
    stability, // Kolom J
    ppm,       // Kolom K
    sesiDur    // Kolom L
  ];
  sheetPerf.appendRow(dataPerf); //[cite: 35]

  return ContentService.createTextOutput("SUCCESS").setMimeType(ContentService.MimeType.TEXT); //[cite: 35]
}

function ambilDataProduksi10Hari(idMesin) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetLog = ss.getSheetByName("log");
  const dataLog = sheetLog.getDataRange().getValues();
  const sheetReff = ss.getSheetByName("reff");
  const dataReff = sheetReff.getDataRange().getValues();
  
  // 1. Ambil Parameter Konfigurasi dari Tabel Reff
  let tglMulaiTagihan = 1;
  let targetBulanan = 0;
  let targetHarianManual = 0;
  for (let j = 1; j < dataReff.length; j++) {
    if (String(dataReff[j][0]) === String(idMesin)) { 
      tglMulaiTagihan = new Date(dataReff[j][14]).getDate(); // Kolom O
      targetBulanan = Number(dataReff[j][12]) || 0;         // Kolom M (Target Minimal)
      targetHarianManual = (Number(dataReff[j][18]) || 0) / 25;
      break;
    }
  }

  // 2. Tentukan Rentang Periode Berjalan
  const sekarang = new Date();
  let tglAwalPeriode = new Date(sekarang.getFullYear(), sekarang.getMonth(), tglMulaiTagihan);
  if (sekarang.getDate() < tglMulaiTagihan) {
    tglAwalPeriode.setMonth(tglAwalPeriode.getMonth() - 1);
  }

  // 3. Kelompokkan bill_harian per tanggal dalam periode ini
  let mapHarian = {};
  for (let i = 1; i < dataLog.length; i++) {
    let tglLog = new Date(dataLog[i][2]); // Kolom C
    if (String(dataLog[i][7]) === String(idMesin) && tglLog >= tglAwalPeriode && tglLog <= sekarang) {
      let key = Utilities.formatDate(tglLog, "GMT+7", "yyyy-MM-dd");
      mapHarian[key] = (mapHarian[key] || 0) + (Number(dataLog[i][6]) || 0); // Kolom G (bill_harian)
    }
  }

  // 4. Hitung Akumulasi (Total Bill Periode) dan Siapkan Label X
  let labelsX = [];
  let dataAkumulasiY = [];
  let runningTotal = 0;
  let sortedDates = Object.keys(mapHarian).sort();
  
  sortedDates.forEach(dateKey => {
    let pemakaianHariIni = mapHarian[dateKey];
    runningTotal += pemakaianHariIni;
    
    let tglObj = new Date(dateKey);
    // Label X: "Tgl (Jumlah Harian)" -> Contoh: "12 Mei (450)"
    let labelX = Utilities.formatDate(tglObj, "GMT+7", "dd MMM") + " (" + pemakaianHariIni + ")";
    
    labelsX.push(labelX);
    dataAkumulasiY.push(runningTotal);
  });

  // Data tambahan untuk widget
  let totalProd = 0, totalBill = 0, billToday = 0;
  if (dataLog.length > 1) {
    for (let i = dataLog.length - 1; i >= 1; i--) {
      if (String(dataLog[i][7]) === String(idMesin)) {
        totalProd = dataLog[i][3];
        totalBill = dataLog[i][4];
        billToday = dataLog[i][6];
        break;
      }
    }
  }

  return {
    labels: labelsX,
    dataBill: dataAkumulasiY,
    target: targetBulanan,
    targetDay: targetHarianManual,
    lifetimeProd: totalProd,
    lifetimeBill: totalBill,
    billHarianLog: billToday
  };
}

 * - Tabel Histori Bawah: Mengambil murni dari tabel 'maintenance'

//////
function ambilDataMaintenance(idMesin) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheetPerf = ss.getSheetByName("performa") || ss.getSheetByName("log"); 
    const sheetMaint = ss.getSheetByName("maintenance"); 
    const sheetLog = ss.getSheetByName("log");
    
    var dataLog = sheetLog.getDataRange().getValues();
    let stability = 0;
    let stabilityHarian = 0; 
    let ppm = 0;
    let jamRate = 0;
    var billCounterTerakhir = 0;
    
    // ========================================================
    // 1. AMBIL DATA COUNTER TERAKHIR (DARI LOG)
    // ========================================================
    for (var i = dataLog.length - 1; i >= 1; i--) {
      if (String(dataLog[i][7]).trim() == String(idMesin).trim()) { 
        billCounterTerakhir = dataLog[i][4]; // Kolom E (Indeks Kolom ke-5)
        break; 
      }
    }
    
    // Ambil tanggal hari ini
    const hariIniStr = Utilities.formatDate(new Date(), "GMT+7", "yyyy-MM-dd");
    
    // ========================================================
    // 2. AMBIL DATA REAL-TIME & HARIAN (UNTUK GRAFIK MAINTENANCE)
    // ========================================================
    if (sheetPerf) {
      const dataPerf = sheetPerf.getDataRange().getValues();
      if (dataPerf.length > 1) {
        let matchCount = 0;
        let sumStability = 0;
        let sumJam = 0;
        let matchCountHarian = 0;
        let sumStabilityHarian = 0;
        
        for (let i = dataPerf.length - 1; i >= 1; i--) {
          if (String(dataPerf[i][1]).trim() === String(idMesin).trim() || String(dataPerf[i][1]).trim() === String(idMesin).trim()) {
            if (matchCount < 3) {
              if (matchCount === 0) {
                ppm = Number(dataPerf[i][10]) || 0; 
                stability = Number(dataPerf[i][9]) || 0;
              }
              sumStability += (Number(dataPerf[i][9]) || 95); 
              sumJam += (Number(dataPerf[i][6]) || 0);      
              matchCount++;
            }
            
            let tglLog = dataPerf[i][0]; 
            if (tglLog instanceof Date) {
              let tglLogStr = Utilities.formatDate(tglLog, "GMT+7", "yyyy-MM-dd");
              if (tglLogStr === hariIniStr) {
                sumStabilityHarian += (Number(dataPerf[i][9]) || 95);
                matchCountHarian++;
              }
            }
          }
        }
        if (matchCount > 0) {
          stability = Math.round(sumStability / matchCount);
          jamRate = ((sumJam / matchCount) * 100).toFixed(1);
        }
        if (matchCountHarian > 0) {
          stabilityHarian = Math.round(sumStabilityHarian / matchCountHarian);
        } else {
          stabilityHarian = stability || 100;
        }
      }
    }

    // ========================================================
    // 3. AMBIL DATA HISTORI (MURNI DARI TABEL MAINTENANCE)
    // ========================================================
    let gabunganHistori = [];
    
    if (sheetMaint) {
      const dataMaint = sheetMaint.getDataRange().getValues();
      if (dataMaint.length > 1) {
        let count = 0;
        
        for (let j = dataMaint.length - 1; j >= 1; j--) {
          // Pengecekan ID Mesin di Kolom A (Index 0)
          if (String(dataMaint[j][1]).trim() === String(idMesin).trim()) {
            
            let statusTiket = String(dataMaint[j][10]).toUpperCase().trim() || "OPEN"; // Kolom K (Index 10)
            let tglTampilan = "-";
            let picAktivitas = "-";
            let jenisTindakan = "-";
            let deskripsiAktivitas = "-";
            
            if (statusTiket === "OPEN") {
              tglTampilan = dataMaint[j][2] ? Utilities.formatDate(new Date(dataMaint[j][2]), "GMT+7", "dd-MM-yyyy HH:mm") : "-";
              picAktivitas = dataMaint[j][3] || "Operator"; 
              jenisTindakan = "PANGGILAN URGENT";
              deskripsiAktivitas = "🚨 KELUHAN: " + (dataMaint[j][4] || "-"); 
            } else {
              tglTampilan = dataMaint[j][5] ? Utilities.formatDate(new Date(dataMaint[j][5]), "GMT+7", "dd-MM-yyyy HH:mm") : "-"; 
              picAktivitas = dataMaint[j][7] || "Teknisi"; 
              jenisTindakan = (dataMaint[j][8] || "PERBAIKAN").toUpperCase(); 
              deskripsiAktivitas = "🛠️ " + (dataMaint[j][9] || "-"); 
            }
            
            // STRUKTUR SUPER GABUNGAN: 
            // Memasukkan semua nama variabel yang dicari oleh halaman lama & baru sekaligus
            gabunganHistori.push({
              waktu: tglTampilan,
              status: statusTiket,
              keterangan: deskripsiAktivitas,
              
              // Kunci variabel untuk halaman service.html yang baru
              operator: picAktivitas,
              kategori: jenisTindakan,
              
              // Kunci variabel asli untuk halaman maintenance.html yang lama
              pic: picAktivitas,
              aktivitas: jenisTindakan
            });
            
            count++;
            if (count >= 20) break; // Kuota dinaikkan agar tabel histori lama terisi penuh
          }
        }
      }
    }

    // ========================================================
    // 4. RETURN PAKET DATA GLOBAL SUPER LENGKAP
    // ========================================================
    return {
      success: true,
      billCounter: billCounterTerakhir,      // Dicari oleh panel service
      bill_counter: billCounterTerakhir,     // Duplikasi fail-safe
      stability: stability || 100,           // Dicari grafik radar/gauge
      stabilityHarian: stabilityHarian,       // Dicari grafik harian
      ppm: ppm,                              // Dicari gauge PPM
      jamRate: jamRate || 0,                 // Dicari gauge Jam Kerja
      histori: gabunganHistori               // Array tunggal yang menyelamatkan kedua halaman!
    };
    
  } catch (err) {
    return { success: false, error: err.toString() };
  }
}

function hitungTagihanBulanan(idMesin) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const sheetLog = ss.getSheetByName("log");
  const sheetReff = ss.getSheetByName("reff");
  const dataReff = sheetReff.getDataRange().getValues();
  
  let config = {};
  // Mengambil konfigurasi dari tabel reff 
  for (let i = 1; i < dataReff.length; i++) {
    if (String(dataReff[i][0]).trim() == String(idMesin).trim()) {
      config = { 
        hTgl: new Date(dataReff[i][14]).getDate(), // Tanggal mulai di kolom O 
        min: Number(dataReff[i][12]) || 0,        // Minimal kuota di kolom M 
        cost: Number(dataReff[i][13]) || 0         // Cost per clik di kolom N 
      };
      break;
    }
  }
  
  if (!config.hTgl) return { success: false };

  // Menentukan rentang tanggal periode berjalan 
  const skr = new Date();
  let tglM = new Date(skr.getFullYear(), skr.getMonth(), config.hTgl);
  if (skr.getDate() < config.hTgl) tglM.setMonth(tglM.getMonth() - 1);
  const tglA = new Date(tglM);
  tglA.setMonth(tglA.getMonth() + 1);

  // Menghitung akumulasi billing_harian dari tabel log (Kolom G) [cite: 94, 95]
  const logs = sheetLog.getDataRange().getValues();
  let totalBillPeriode = 0;
  for (let j = 1; j < logs.length; j++) {
    let tglLog = new Date(logs[j][2]); // Kolom C [cite: 94]
    if (String(logs[j][7]).trim() == String(idMesin).trim() && tglLog >= tglM && tglLog < tglA) {
      totalBillPeriode += (Number(logs[j][6]) || 0); // Akumulasi dari bill_harian 
    }
  }

  // Menghitung cliks yang harus dibayar 
  let selisih = totalBillPeriode - config.min;
  let totalRupiah = selisih > 0 ? selisih * config.cost : 0;

  return {
    success: true,
    total: totalRupiah,
    qty: totalBillPeriode,
    min: config.min,
    cost: config.cost,
    periode: Utilities.formatDate(tglM, "GMT+7", "dd MMM") + " - " + Utilities.formatDate(tglA, "GMT+7", "dd MMM yyyy")
  };
}

/**
 * 1. Fungsi Matematika Hitung Jarak (Haversine Formula) - Satuan METER
 */
function hitungJarakHaversine(lat1, lon1, lat2, lon2) {
  try {
    const R = 6371000; // Jari-jari bumi dalam meter
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLon = (lon2 - lon1) * Math.PI / 180;
    const a = 
      Math.sin(dLat/2) * Math.sin(dLat/2) +
      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) * Math.sin(dLon/2) * Math.sin(dLon/2);
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1-a));
    return R * c; 
  } catch (err) {
    return 999999; // Fail-safe jika error jarak dibuat jauh agar ditolak
  }
}

/**
 * 2. Fungsi Memeriksa Status Kolom Q pada Tabel "reff"
 */
function cekAtauSimpanKoordinatMesin(idMesin, koordinatBaru) {
  try {
    var ss = SpreadsheetApp.getActiveSpreadsheet();
    var sheetReff = ss.getSheetByName("reff");
    if (!sheetReff) return { status: "ERROR", pesan: "Tabel 'reff' tidak ditemukan" };
    
    var data = sheetReff.getDataRange().getValues();
    var idxKolomQ = 16; // Kolom Q (A=0, B=1, ... Q=16)
    
    for (var i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim() === String(idMesin).trim()) {
        var koordinatLama = data[i][idxKolomQ];
        
        // Skenario 1: Jika diperintahkan untuk menyimpan data baru ke kolom Q yang kosong
        if (koordinatBaru && (!koordinatLama || koordinatLama.toString().trim() === "")) {
          sheetReff.getRange(i + 1, idxKolomQ + 1).setValue(koordinatBaru);
          SpreadsheetApp.flush(); // Paksa Google Sheets untuk langsung mengunci data detik itu juga
          return { status: "TERISI", data: koordinatBaru };
        }
        
        // Skenario 2: Murni pengecekan awal di awal klik tombol panggil
        if (!koordinatLama || koordinatLama.toString().trim() === "") {
          return { status: "KOSONG" };
        } else {
          return { status: "TERISI", data: koordinatLama.toString() };
        }
      }
    }
    return { status: "NOT_FOUND", pesan: "ID Mesin tidak ditemukan" };
  } catch (err) {
    return { status: "ERROR", pesan: err.toString() };
  }
}




/**
 * 3. Fungsi Validasi Jarak Teknisi (Maksimal 200 Meter)
 */
function validasiJarakTeknisi(idMesin, latTeknisi, lonTeknisi) {
  try {
    const ss = SpreadsheetApp.getActiveSpreadsheet();
    const sheetReff = ss.getSheetByName("reff");
    const data = sheetReff.getDataRange().getValues();
    const idxKolomQ = 16;
    
    let koordinatMesin = "";
    for (let i = 1; i < data.length; i++) {
      if (String(data[i][0]).trim() === String(idMesin).trim()) {
        koordinatMesin = data[i][idxKolomQ];
        break;
      }
    }
    
    // Jika koordinat acuan mesin belum diset, anggap lolos (atau preventif awal)
    if (!koordinatMesin || koordinatMesin.toString().trim() === "") {
      return { lolos: true, catatan: "Koordinat acuan mesin belum terdaftar." };
    }
    
    // Pecah string "lat, lon" dari kolom Q
    let parts = koordinatMesin.toString().split(",");
    let latMesin = parseFloat(parts[0].trim());
    let lonMesin = parseFloat(parts[1].trim());
    
    // Hitung jarak murni menggunakan Haversine
    let jarakMeter = hitungJarakHaversine(latTeknisi, lonTeknisi, latMesin, lonMesin);
    
    if (jarakMeter <= 200) {
      return { lolos: true, jarak: Math.round(jarakMeter) };
    } else {
      return { lolos: false, jarak: Math.round(jarakMeter) };
    }
  } catch (err) {
    return { lolos: false, catatan: "Gagal menghitung geofence: " + err.toString() };
  }
}
